\newenvironment{chapterintro}[4]{
% PŘED TEXTEM:
    % Dostaň se na levou stranu
    \clearpage
    \ifodd\c@page
        \hfill
        \clearpage
    \fi
    % Levá (sudá) strana
    \thispagestyle{empty}
    \xglobal\colorlet{accentColor}{#3}
    \pagecolor{accentColor}
    \tikz[remember picture,overlay] \node[inner sep=0pt] at (current page.center) {#4};
    % Pravá (lichá) strana
    \chapter{#1}
    \label{#2}
    \thispagestyle{empty}
    \pagecolor{accentColor!50!white}
}{
% ZA TEXTEM
    % Jdi na následující stranu a vymaž barvu stránky
    \clearpage
    \nopagecolor
}

\titleformat{\chapter}[display]
    {\LARGE\sffamily\lsstyle}
    {\chaptertitlename\ \thechapter}
    {0pt}
    {\fontsize{40pt}{40pt}\scshape\lsstyle\raggedright}[]

\pretocmd{\section}{\clearpage}{}{}
\titleformat{\section}[hang]
    {\Large\bfseries\sffamily}
    {\color{accentColor!40!black}\thesection\;\color{accentColor!70!black}\textbullet\;}
    {0pt}
    {}[]

\titleformat{\subsection}[hang]
    {\large\bfseries\scshape\sffamily}
    {\color{accentColor!70!darkgray}\thesubsection}
    {5mm}
    {\color{darkgray}}[]
